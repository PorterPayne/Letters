    <script>
      const targetWord = 'CUTIE';
      let currentRow = 1;
      
      function submitWord() {
        const wordInput = document.getElementById('word-input');
        const word = wordInput.value.toUpperCase();
        
        if (word.length !== 5) {
          alert('Please enter a 5-letter word');
          return;
        }
        
        const wordleBoxes = document.querySelectorAll(`#row-${currentRow} .wordle-box`);
        
        for (let i = 0; i < 5; i++) {
          const box = wordleBoxes[i];
          const letter = word[i];
          
          box.textContent = letter;
          
          if (letter === targetWord[i]) {
            box.classList.add('correct');
          } else if (targetWord.includes(letter)) {
            box.classList.add('close');
          } else {
            box.classList.add('incorrect');
          }
        }
        
        currentRow++;
        
        if (currentRow > 6) {
          alert('Game over!');
        }
        
        if (
          wordleBoxes[0].textContent === targetWord[0] &&
          wordleBoxes[1].textContent === targetWord[1] &&
          wordleBoxes[2].textContent === targetWord[2] &&
          wordleBoxes[3].textContent === targetWord[3] &&
          wordleBoxes[4].textContent === targetWord[4]
        ) {
          //alert('Congratulations! You won!');
          currentRow = 7;
        }
        
        wordInput.value = '';
      }
    </script>
