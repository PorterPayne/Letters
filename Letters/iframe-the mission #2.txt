	<script>
		const container = document.querySelector('.container');
		const startDate = new Date();
		const endDate = new Date('2025-05-15'); // Change this to the desired end date
		const totalDays = Math.round((endDate - startDate) / (1000 * 3600 * 24));

		for (let i = 0; i < 715; i++) {
			const square = document.createElement('div');
			square.className = 'square';
			if (i < totalDays) {
				square.className += ' red';
			}
			container.appendChild(square);
		}

		// Update the squares every day
		setInterval(() => {
			const currentDate = new Date();
			const daysUntilEnd = Math.round((endDate - currentDate) / (1000 * 3600 * 24));
			const squares = document.querySelectorAll('.square');
			squares.forEach((square, index) => {
				if (index < daysUntilEnd) {
					square.classList.add('red');
				} else {
					square.classList.remove('red');
				}
			});
		}, 1000 * 3600 * 24); // Update every day
	</script>
